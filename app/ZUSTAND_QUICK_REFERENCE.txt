## Minimum Setup (Copy-Paste These 3 Things)

### 1. Import the Hook
```tsx
import { useEvents } from "@/hooks/useEvents";
import { useCommunities } from "@/hooks/useCommunities";
```

### 2. Use in Component
```tsx
const { events, isRefreshing, refresh } = useEvents();
const { communities, isRefreshing: commRefresh, refresh: refCom } = useCommunities();
```

### 3. Add to ScrollView
```tsx
<ScrollView
  refreshControl={<PullToRefresh 
    isRefreshing={isRefreshing || commRefresh} 
    onRefresh={async () => {
      await Promise.all([refresh(), refCom()]);
    }} 
  />}
>
  {/* Your content */}
</ScrollView>
```

## Hooks API

### useEvents()
```tsx
const {
  events,           // Event[] - the data
  isLoading,        // boolean - initial load
  isRefreshing,     // boolean - user pulling to refresh
  error,            // string | null - error message
  refresh,          // async () => Promise<void> - force refresh
} = useEvents(autoFetch?: boolean);
```

### useCommunities()
```tsx
const {
  communities,      // Community[] - the data
  isLoading,        // boolean - initial load
  isRefreshing,     // boolean - user pulling to refresh
  error,            // string | null - error message
  refresh,          // async () => Promise<void> - force refresh
} = useCommunities(autoFetch?: boolean);
```

## Store API (Advanced)

### useEventStore()
```tsx
const {
  events,                         // Event[] 
  eventDetails,                   // { [id]: Event }
  isLoading,                      // boolean
  lastFetchTime,                  // number | null
  
  fetchEvents,                    // (force?) => Promise
  fetchEventById,                 // (id, force?) => Promise<Event|null>
  setEvents,                      // (events) => void
  setEventDetails,                // (id, event) => void
  clear,                          // () => void
} = useEventStore();
```

## Common Usage Patterns

### Pattern 1: Basic Load
```tsx
const { events, isRefreshing, refresh } = useEvents();
<ScrollView refreshControl={<PullToRefresh isRefreshing={isRefreshing} onRefresh={refresh} />}>
  {events.map(e => <EventCard key={e.id} event={e} />)}
</ScrollView>
```

### Pattern 2: Multiple Data
```tsx
const { events, refresh: r1, isRefreshing: ir1 } = useEvents();
const { communities, refresh: r2, isRefreshing: ir2 } = useCommunities();
const refresh = () => Promise.all([r1(), r2()]);
<ScrollView refreshControl={<PullToRefresh isRefreshing={ir1 || ir2} onRefresh={refresh} />}>
```

### Pattern 3: Manual Load
```tsx
const { events, refresh } = useEvents(false); // autoFetch: false
<Button title="Load" onPress={refresh} />
```

### Pattern 4: Detail Page
```tsx
const { eventDetails, fetchEventById } = useEventStore();
useEffect(() => fetchEventById(id), [id]);
const event = eventDetails[id];
```

### Pattern 5: After Mutation
```tsx
const { events, setEvents } = useEventStore();
// After updating an event:
const updated = events.map(e => e.id === id ? {...e, name} : e);
setEvents(updated); // Updates cache immediately
```

## Troubleshooting

### "Data not updating after changes"
→ Call `setEvents()` to update cache
```tsx
const { setEvents, events } = useEventStore();
setEvents([...newEvents]);
```

### "Still showing old data after logout"
→ Clear all stores on logout
```tsx
useEventStore().clear();
useCommunityStore().clear();
```

### "Data is stale, need fresh"
→ Call `refresh()` to force new API call
```tsx
const { refresh } = useEvents();
await refresh(); // Always fetches fresh
```

### "Two screens showing different data"
→ Both should use same store - verify import path
